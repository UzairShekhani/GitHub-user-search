Create custom cursor using simple HTML CSS and JS

Menus play a crucial role in enhancing user experience and navigation on a website. Using HTML, CSS, and JavaScript, we can create dynamic and interactive menus that provide users with seamless access to various sections and features of a website. HTML defines the structure of the menu, CSS allows us to style and customize its appearance to match the website's design, while JavaScript adds interactivity, such as dropdown functionality, hover effects, and mobile responsiveness.

    <script>
      let folder = document.querySelector(".folder");
      document.addEventListener("click", (e) => {
        if (e.target == folder || folder.contains(e.target)) {
          if (folder.classList.contains("close")) {
            folder.classList.toggle("close");
            folder.classList.toggle("open");
          }
        } else {
          if (folder.classList.contains("open")) {
            folder.classList.toggle("close");
            folder.classList.toggle("open");
          }
        }
      });
    </script>

    <script>
      let menu = document.querySelector(".menu");
      let intiHeight = menu.offsetHeight;
      let initWidth = menu.offsetWidth;
      menu.style.display = "none";
      window.addEventListener("contextmenu", (e) => {
        e.preventDefault();

        let positionX = e.clientX;
        let positionY = e.clientY;
        let wHeight = window.innerHeight;
        let wWidth = window.innerWidth;

        let menuHeight = menu.offsetHeight || intiHeight;
        let menuWidth = menu.offsetWidth || initWidth;

        let availableHeight = wHeight - positionY;
        let availableWidth = wWidth - positionX;

        if (availableHeight >= menuHeight) {
          menu.style.top = positionY + "px";
          menu.style.bottom = "auto";
        } else {
          menu.style.bottom = wHeight - positionY + "px";
          menu.style.top = "auto";
          if (positionY < menuHeight) {
            menu.style.top = "0px";
            menu.style.bottom = "auto";
          }
        }

        if (availableWidth >= menuWidth) {
          menu.style.left = positionX + "px";
          menu.style.right = "auto";
        } else {
          menu.style.right = wWidth - positionX + "px";
          menu.style.left = "auto";
        }
        menu.style.display = "block";
      });

      window.addEventListener("click", () => {
        menu.style.display = "none";
      });
    </script>
