Different types of input styles

When it comes to designing a form in HTML, input tags play a crucial role in capturing user input. However, styling them effectively can enhance the form's overall look and feel, making it more attractive and user-friendly. In this blog post, we'll explore three simple and visually appealing designs for input tags that you can easily implement in your HTML forms.

The first design is a flat-style input tag that uses simple outline with curved border and color variations to highlight the focused input field. This design is clean and minimal, making it perfect for modern and professional websites.

The second design input tag, which features a bottom border and background with sharp edges, is a popular choice for creating a clean and modern look in HTML forms. This design is often referred to as the "bottom border input" or "underline input" because it only adds a border and background at the bottom of the input field

    <script>
      let tagBox = document.querySelector(".tag-input");

      function findWidth(val) {
        let tempElement = document.createElement("span");
        document.body.appendChild(tempElement);
        tempElement.style.display = "block";
        tempElement.style.width = "fit-content";
        tempElement.style.fontFamily = "poppins";
        tempElement.style.fontSize = "14px";
        tempElement.innerHTML = val;
        let width = tempElement.offsetWidth;
        tempElement.remove();
        return Number(width) + 10;
      }

      document.addEventListener("keydown", (e) => {
        let key = e.key;
        if (key == "," || key == "Enter") {
          e.preventDefault();
          if (e.target.parentElement.className == "inputf") {
            let parent = e.target.parentElement;
            parent.className = "inputf tag";

            let input = e.target;
            input.style.width = findWidth(input.value) + "px";

            let newTag = document.createElement("span");
            newTag.className = "inputf";
            let newInput = document.createElement("input");
            newInput.className = "input";
            newTag.appendChild(newInput);
            newTag.insertAdjacentHTML(
              "beforeend",
              '<span class="cancel material-icons-outlined">highlight_off</span>'
            );
            tagBox.appendChild(newTag);
            newInput.focus();
          }
        }
      });

      document.addEventListener("input", (e) => {
        let parent = e.target.parentElement;
        if (parent.classList.contains("tag")) {
          let taggedInput = e.target;
          taggedInput.style.width = findWidth(taggedInput.value) + "px";
        }
      });

      document.addEventListener("click", (e) => {
        if (e.target.classList.contains("cancel")) {
          let parent = e.target.parentElement;
          parent.nextElementSibling.querySelector(".input").focus();
          parent.remove();
        }
      });
    </script>
