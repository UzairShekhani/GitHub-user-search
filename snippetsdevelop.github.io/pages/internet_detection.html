Detect Internet Connectivity using JavaScript HTML CSS

In today's interconnected world, having a reliable internet connection is essential. But what if you could enhance your website or application by dynamically detecting internet connectivity? We'll dive into the fascinating world of web development as we explore how to detect internet connectivity using the powerful trio of HTML, CSS, and JavaScript. We'll learn how to leverage HTML's offline capabilities, style offline and online states with CSS, and implement JavaScript to detect and respond to changes in internet connectivity.

    <script>
      let band = document.querySelector(".band");
      let box = document.querySelector(".box");
      let boxBtn = document.querySelector("#btn");
      let icon = document.querySelector(".icon");
      let text = document.querySelector(".text");
      window.addEventListener("offline", () => {
        band.style.background = "rgb(0,0,0)";
        icon.style.color = "rgb(255,255,255,0.5)";
        icon.innerHTML = "wifi_off";
        text.innerHTML = "You are currently offline";
        band.innerHTML = "No Internet Connection";
        boxBtn.style.display = "block";
        if (band.className == "band") {
          band.classList.toggle("active");
        }
        if (box.className == "box") {
          box.classList.toggle("active");
        }
      });
      window.addEventListener("online", () => {
        band.style.background = "rgb(18,128,255)";
        icon.style.color = "rgb(109,230,142)";
        icon.innerHTML = "wifi";
        text.innerHTML = "Your connection is restored";
        band.innerHTML = "Connected to Internet";
        boxBtn.style.display = "none";
        if (band.className == "band active") {
          setTimeout(() => {
            band.classList.toggle("active");
          }, 3000);
        }
        if (box.className == "box active") {
          setTimeout(() => {
            box.classList.toggle("active");
          }, 3000);
        }
      });
    </script>
