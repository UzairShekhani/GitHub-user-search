Create a Stunning Progress Bar with Spike Design using HTML, CSS, and JS

Progress bars play a crucial role in user experience by providing visual feedback on the completion status of a task or process. They help users understand the progress made, set expectations, and reduce uncertainty. Whether it's indicating the loading progress of a web page, the completion of a form, or the progress of a file upload, progress bars enhance usability and keep users engaged. By mastering the creation of a captivating circular progress bar, readers can enhance their web development skills and create more intuitive and visually appealing user interfaces.

      let spikes = document.querySelector(".spikes");
      for (var i = 0; i < 360; i += 3) {
        let spike = document.createElement("span");
        spike.className = "spike";
        spike.style.transform = `rotate(${i}deg) translateY(-135px)`;
        spikes.appendChild(spike);
      }

      let allspikes = document.querySelectorAll(".spike");
      let n = allspikes.length;
      i = 0;
      const interval = setInterval(() => {
        allspikes[i].className = "spike active";
        let percent = Math.ceil(((i + 1) / n) * 100);
        document.querySelector(".value").innerHTML = percent + "%";
        i = i + 1;
        if (i == n) {
          document.querySelector(".value").innerHTML = "0%";
          i = 0;
          allspikes.forEach((spike) => {
            spike.className = "spike";
          });
        }
      }, 200);
    </script>
