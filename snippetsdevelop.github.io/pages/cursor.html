Create custom cursor using simple HTML CSS and JS

Custom cursors provide an incredible opportunity to infuse your website with creativity and personality. With HTML, CSS, and JS, you have the power to design a cursor that aligns perfectly with your brand or theme. Imagine having a cursor that morphs into various shapes, animates with eye-catching effects, or interacts with elements on the page. These customizations can leave a lasting impression on your visitors, making your website more engaging and immersive.

    <script>
      var cursor = document.querySelectorAll(".cursor");
      document.addEventListener("mousemove", (event) => {
        cursor.forEach((item) => {
          item.style.top = event.clientY + "px";
          item.style.left = event.clientX + "px";
        });
      });

      var links = document.querySelectorAll("a");
      links.forEach((item) => {
        item.addEventListener("mouseenter", () => {
          cursor.forEach((point) => {
            point.classList.add("hover");
          });
        });

        item.addEventListener("mouseleave", () => {
          cursor.forEach((point) => {
            point.classList.remove("hover");
          });
        });
      });
    </script>
